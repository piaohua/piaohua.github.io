Quickly connecting to redis

1
2
3
4
5

import redis
r = redis.Redis(host='localhost', port=6379, db=0)
r.ping()
r.set('foo', 'bar')
r.get('foo')

Connection Pools

1
2
3
4

import redis
pool = redis.ConnectionPool(host='localhost', port=6379, db=0)
r = redis.Redis(connection_pool=pool)
r.ping()
